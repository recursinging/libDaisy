228  // init the individual drivers
229  for(int d = 0; d < numDrivers; d++)
230  {
231  const uint8_t address = PCA9685_I2C_BASE_ADDRESS | addresses_[d];
232  uint8_t buffer[2];
233  buffer[0] = PCA9685_MODE1;
234  buffer[1] = 0x00;
235  i2c_.TransmitBlocking(address, buffer, 2, 1);
236  System::Delay(20);
237  buffer[0] = PCA9685_MODE1;
238  buffer[1] = 0x00;
239  i2c_.TransmitBlocking(address, buffer, 2, 1);
240  System::Delay(20);
241  buffer[0] = PCA9685_MODE1;
242  // auto increment on
243  buffer[1] = 0b00100000;
244  i2c_.TransmitBlocking(address, buffer, 2, 1);
245  System::Delay(20);
