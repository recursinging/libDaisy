194  // If we write "ChildType::DrawPixel(x2, y2, on);", we end up with
195  // all sorts of weird compiler errors when the Child class is a template
196  // class. The only way around this is to use this very verbose syntax:
197  ((ChildType*)(this))->ChildType::DrawPixel(x2, y2, on);
198 
199  while((x1 != x2) || (y1 != y2))
200  {
201  ((ChildType*)(this))->ChildType::DrawPixel(x1, y1, on);
202  error2 = error * 2;
203  if(error2 > -deltaY)
204  {
205  error -= deltaY;
206  x1 += signX;
207  }
208

210  {
211  error += deltaX;
212  y1 += signY;
213  }

293  t_y = 0;

303  t_synx = start_y * -t_x;
304  t_exy = end_x * t_y;

306  t_exny = end_x * -t_y;
307  t_eynx = end_y * -t_x;
308

320  d3 = (t_sxny - t_syx < 0 && t_exny - t_eyx > 0);
