141  }

146  // transitioning?
147  if(buttonStates_[id] - 1 < timeout_)

154  else if(retriggerTimeoutMs_ > 0)

156  const auto timeSincePress

158  if(timeSincePress >= retriggerTimeoutMs_)
159  {
