156 void Parse(uint8_t byte)
157 {
158 switch(pstate_)
159 {
160 case ParserEmpty:
161 // check byte for valid Status Byte
162 if(byte & kStatusByteMask)
163 {
164 // Get MessageType, and Channel
165 incoming_message_.channel = byte & kChannelMask;
166 incoming_message_.type = static_cast<MidiMessageType>(
167 (byte & kMessageMask) >> 4);
168
169 // Validate, and move on.
170 if(incoming_message_.type < MessageLast)
171 {
172 pstate_ = ParserHasStatus;
173 // Mark this status byte as running_status
174 running_status_ = incoming_message_.type;
175
176 if(running_status_ == SystemCommon)
177 {
178 incoming_message_.channel = 0;
179 //system real time = 1111 1xxx
180 if(byte & 0x08)
181 {
182 incoming_message_.type = SystemRealTime;
183 running_status_ = SystemRealTime;
184 incoming_message_.srt_type
185 = static_cast<SystemRealTimeType>(
186 byte & kSystemRealTimeMask);
187
188 //short circuit to start
189 pstate_ = ParserEmpty;
190 event_q_.Write(incoming_message_);
191 }
192 //system common
193 else
194 {
195 incoming_message_.sc_type
196 = static_cast<SystemCommonType>(byte
197 & 0x07);
198 //sysex
199 if(incoming_message_.sc_type == SystemExclusive)
200 {
201 pstate_ = ParserSysEx;
202 incoming_message_.sysex_message_len = 0;
203 }
204 //short circuit
205 else if(incoming_message_.sc_type > SongSelect)
206 {
207 pstate_ = ParserEmpty;
208 event_q_.Write(incoming_message_);
209 }
210 }
211 }
212 }
213 // Else we'll keep waiting for a valid incoming status byte
214 }
215 else
216 {
217 // Handle as running status
218 incoming_message_.type = running_status_;
219 incoming_message_.data[0] = byte & kDataByteMask;
220 pstate_ = ParserHasData0;
221 }
222 break;
223 case ParserHasStatus:
224 if((byte & kStatusByteMask) == 0)
225 {
226 incoming_message_.data[0] = byte & kDataByteMask;
227 if(running_status_ == ChannelPressure
228 || running_status_ == ProgramChange
229 || incoming_message_.sc_type == MTCQuarterFrame
230 || incoming_message_.sc_type == SongSelect)
231 {
232 //these are just one data byte, so we short circuit back to start
233 pstate_ = ParserEmpty;
234 event_q_.Write(incoming_message_);
235 }
236 else
237 {
238 pstate_ = ParserHasData0;
239 }
240
241 //ChannelModeMessages (reserved Control Changes)
242 if(running_status_ == ControlChange
243 && incoming_message_.data[0] > 119)
244 {
245 incoming_message_.type = ChannelMode;
246 running_status_ = ChannelMode;
247 incoming_message_.cm_type
248 = static_cast<ChannelModeType>(
249 incoming_message_.data[0] - 120);
250 }
251 }
252 else
253 {
254 // invalid message go back to start ;p
255 pstate_ = ParserEmpty;
256 }
257 break;
258 case ParserHasData0:
259 if((byte & kStatusByteMask) == 0)
260 {
261 incoming_message_.data[1] = byte & kDataByteMask;
262
263 //velocity 0 NoteOns are NoteOffs
264 if(running_status_ == NoteOn
265 && incoming_message_.data[1] == 0)
266 {
267 incoming_message_.type = running_status_ = NoteOff;
268 }
269
270 // At this point the message is valid, and we can add this MidiEvent to the queue
271 event_q_.Write(incoming_message_);
272 }
273 // Regardless, of whether the data was valid or not we go back to empty
274 // because either the message is queued for handling or its not.
275 pstate_ = ParserEmpty;
276 break;
277 case ParserSysEx:
278 // end of sysex
279 if(byte == 0xf7)
280 {
281 pstate_ = ParserEmpty;
282 event_q_.Write(incoming_message_);
283 }
284 else if(incoming_message_.sysex_message_len < SYSEX_BUFFER_LEN)
285 {
286 incoming_message_
287 .sysex_data[incoming_message_.sysex_message_len]
288 = byte;
289 incoming_message_.sysex_message_len++;
290 }
291 break;
292 default: break;
293 }
294 }
