155 void Parse(uint8_t byte)
156 {
157 switch(pstate_)
158 {
159 case ParserEmpty:
160 // check byte for valid Status Byte
161 if(byte & kStatusByteMask)
162 {
163 // Get MessageType, and Channel
164 incoming_message_.channel = byte & kChannelMask;
165 incoming_message_.type = static_cast<MidiMessageType>(
166 (byte & kMessageMask) >> 4);
167
168 // Validate, and move on.
169 if(incoming_message_.type < MessageLast)
170 {
171 pstate_ = ParserHasStatus;
172 // Mark this status byte as running_status
173 running_status_ = incoming_message_.type;
174
175 if(running_status_ == SystemCommon)
176 {
177 incoming_message_.channel = 0;
178 //system real time = 1111 1xxx
179 if(byte & 0x08)
180 {
181 incoming_message_.type = SystemRealTime;
182 running_status_ = SystemRealTime;
183 incoming_message_.srt_type
184 = static_cast<SystemRealTimeType>(
185 byte & kSystemRealTimeMask);
186
187 //short circuit to start
188 pstate_ = ParserEmpty;
189 event_q_.Write(incoming_message_);
190 }
191 //system common
192 else
193 {
194 incoming_message_.sc_type
195 = static_cast<SystemCommonType>(byte
196 & 0x07);
197 //sysex
198 if(incoming_message_.sc_type == SystemExclusive)
199 {
200 pstate_ = ParserSysEx;
201 incoming_message_.sysex_message_len = 0;
202 }
203 //short circuit
204 else if(incoming_message_.sc_type > SongSelect)
205 {
206 pstate_ = ParserEmpty;
207 event_q_.Write(incoming_message_);
208 }
209 }
210 }
211 }
212 // Else we'll keep waiting for a valid incoming status byte
213 }
214 else
215 {
216 // Handle as running status
217 incoming_message_.type = running_status_;
218 incoming_message_.data[0] = byte & kDataByteMask;
219 pstate_ = ParserHasData0;
220 }
221 break;
222 case ParserHasStatus:
223 if((byte & kStatusByteMask) == 0)
224 {
225 incoming_message_.data[0] = byte & kDataByteMask;
226 if(running_status_ == ChannelPressure
227 || running_status_ == ProgramChange
228 || incoming_message_.sc_type == MTCQuarterFrame
229 || incoming_message_.sc_type == SongSelect)
230 {
231 //these are just one data byte, so we short circuit back to start
232 pstate_ = ParserEmpty;
233 event_q_.Write(incoming_message_);
234 }
235 else
236 {
237 pstate_ = ParserHasData0;
238 }
239
240 //ChannelModeMessages (reserved Control Changes)
241 if(running_status_ == ControlChange
242 && incoming_message_.data[0] > 119)
243 {
244 incoming_message_.type = ChannelMode;
245 running_status_ = ChannelMode;
246 incoming_message_.cm_type
247 = static_cast<ChannelModeType>(
248 incoming_message_.data[0] - 120);
249 }
250 }
251 else
252 {
253 // invalid message go back to start ;p
254 pstate_ = ParserEmpty;
255 }
256 break;
257 case ParserHasData0:
258 if((byte & kStatusByteMask) == 0)
259 {
260 incoming_message_.data[1] = byte & kDataByteMask;
261
262 //velocity 0 NoteOns are NoteOffs
263 if(running_status_ == NoteOn
264 && incoming_message_.data[1] == 0)
265 {
266 incoming_message_.type = running_status_ = NoteOff;
267 }
268
269 // At this point the message is valid, and we can add this MidiEvent to the queue
270 event_q_.Write(incoming_message_);
271 }
272 // Regardless, of whether the data was valid or not we go back to empty
273 // because either the message is queued for handling or its not.
274 pstate_ = ParserEmpty;
275 break;
276 case ParserSysEx:
277 // end of sysex
278 if(byte == 0xf7)
279 {
280 pstate_ = ParserEmpty;
281 event_q_.Write(incoming_message_);
282 }
283 else if(incoming_message_.sysex_message_len < SYSEX_BUFFER_LEN)
284 {
285 incoming_message_
286 .sysex_data[incoming_message_.sysex_message_len]
287 = byte;
288 incoming_message_.sysex_message_len++;
289 }
290 break;
291 default: break;
292 }
293 }
