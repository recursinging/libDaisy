71 BlockingSpiTransport::Result
72 Transmit(uint8_t* buff, size_t size, uint32_t wait_us)
73 {
74 if(wait_us > 0)
75 {
76 uint32_t ts = System::GetUs();
77 // Since this is a transaction which requires a delay
78 // we check if enough time has elapsed, if not, we simply
79 // return "OK".
80 if(ts < next_tx_)
81 {
82 // Check if the clock rolled over, the max wait time is 20*40us
83 if((next_tx_ - ts) > (20 * 40))
84 {
85 next_tx_ = ts + wait_us;
86 return Result::OK;
87 }
88 else
89 {
90 return Result::OK;
91 }
92 }
93 }
94
95 if(spi_.BlockingTransmit(buff, size) == SpiHandle::Result::ERR)
96 {
97 return Result::ERR;
98 }
99
100 if(wait_us > 0)
101 {
102 // Reset next_tx timestamp
103 next_tx_ = System::GetUs() + wait_us;
104 }
105 return Result::OK;
106 }
107
108 BlockingSpiTransport::Result
109 TransmitAndReceive(uint8_t* tx_buff, uint8_t* rx_buff, size_t size)
110 {
111 if(spi_.BlockingTransmitAndReceive(tx_buff, rx_buff, size)
112 == SpiHandle::Result::ERR)
113 {
114 return Result::ERR;
115 }
116
117 return Result::OK;
118 }

235 Result Init(Config config)
236 {
237 transport_.Init(config.transport_config);
238
239 // First, let's verify the SPI comms, and chip presence. The DEVID register
240 // is a fixed, read-only value we can compare against to ensure we're connected.
241 if(ReadRegister(MAX11300_DEVICE_ID) != 0x0424)
242 {
243 return Result::ERR;
244 }
245
246 // Init routine (roughly) as per the datasheet pp. 49
247 // These settings were chosen as best applicable for use in a Eurorack context.
248 // Should the need for more configurability arise, this would be the spot to do it.
249
250 // Setup the device...
251 uint16_t devctl = 0x0000;
252 // 1:0 ADCCTL[1:0] - ADC conversion mode selection = 11: Continuous sweep
253 devctl = devctl | 0x0003;
254 // 3:2 DACCTL[1:0] - DAC mode selection = 01: Immediate Update mode for DAC-configured ports.
255 devctl = devctl | 0x0004;
256 // 5:4 ADCCONV[1:0] - ADC conversion rate selection = 11: ADC conversion rate of 400ksps
257 devctl = devctl | 0x0030;
258 // 6 DACREF - DAC voltage reference selection = 1: Internal reference voltage
259 devctl = devctl | 0x0040;
260 // 7 THSHDN - Thermal shutdown enable = 1: Thermal shutdown function enabled.
261 devctl = devctl | 0x0080;
262 // 10:8 TMPCTL[2:0] - Temperature monitor selection = 001: Internal temperature monitor enabled
263 devctl = devctl | 0x0100;
264 // 11 TMPPER - Temperature conversion time control = 0 (Default)
265 // 12 RS_CANCEL - Temperature sensor series resistor cancellation mode = 0 (Default)
266 // 13 LPEN - Power mode selection = 0 (Default)
267 // 14 BRST - Serial interface burst-mode selection = 1: Contextual address incrementing mode
268 devctl = devctl | 0x4000;
269 // 15 RESET - Soft reset control = 0 (Default)
270
271 // Write the device configuration
272 if(WriteRegister(MAX11300_DEVCTL, devctl) == Result::ERR)
273 {
274 return Result::ERR;
275 }
276 // Verify our configuration was written...
277 if(ReadRegister(MAX11300_DEVCTL) != devctl)
278 {
279 return Result::ERR;
280 }
281
282 // Add a delay as recommended in the datasheet.
283 DelayUs(200);
284
285 // Set all pins to the default high impedance state...
286 for(uint8_t i = 0; i <= Pin::PIN_19; i++)
287 {
288 PinConfig pin_cfg;
289 pin_cfg.Defaults();
290 pin_configurations_[i] = pin_cfg;
291 SetPinConfig(static_cast<Pin>(i));
292 }
293
294 return Result::OK;
295 }

493 Result Update()
494 {
495 // Check first if were ready to TX/RX
496 if(!transport_.Ready())
497 return Result::OK;
498
499 if(dac_pin_count_ > 0)
500 {
501 // This is a burst transaction utilizing the contextual addressing
502 // scheme of the MAX11300. See the datasheet @ pp. 30
503 //
504 // We've prefixed the dac_buffer_ to point at the first dac pin to be written to.
505 // Subsequent DAC pins are written in their absolute order (0-19) if configured as such.
506 // For example:
507 // [1st_dac_pin_addr],[1st_dac_pin_msb],[1st_dac_pin_lsb],[2nd_dac_pin_msb],[2nd_dac_pin_lsb]...
508 //
509 // The size of the transaction is determined by the number of configured dac pins,
510 // plus one byte for the initial pin address.
511 //
512 // The datasheet recommends waiting 80us between DAC updates, in practice the appears to be
513 // per configured DAC pin. Here we inform the transport to wait at least N uS before transmitting
514 // again.
515 size_t tx_size = (dac_pin_count_ * 2) + 1;
516 if(transport_.Transmit(dac_buffer_, tx_size, (dac_pin_count_ * 40))
517 != Transport::Result::OK)
518 {
519 return Result::ERR;
520 }
521 }
522
523 if(adc_pin_count_ > 0)
524 {
525 // Reading ADC pins is a burst transaction approximately the same as the DAC transaction
526 // as described above...
527 size_t size = (adc_pin_count_ * 2) + 1;
528
529 uint8_t tx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
530 tx_buff[0] = adc_buffer_[0];
531 if(transport_.TransmitAndReceive(tx_buff, adc_buffer_, size)
532 != Transport::Result::OK)
533 {
534 adc_buffer_[0] = tx_buff[0];
535 return Result::ERR;
536 }
537 adc_buffer_[0] = tx_buff[0];
538 }
539
540 if(gpo_pin_count_ > 0)
541 {
542 // Writing GPO pins is a single 5 byte transaction, with the first byte being the
543 // the GPO data register, and the subsequent 4 bytes containing the state of the
544 // GPO ports to be written.
545 if(transport_.Transmit(gpo_buffer_, sizeof(gpo_buffer_), 0)
546 != Transport::Result::OK)
547 {
548 return Result::ERR;
549 }
550 }
551
552 if(gpi_pin_count_ > 0)
553 {
554 // Reading GPI pins is a single, 5 byte, full-duplex transaction with the first
555 // and only TX byte being the GPI register.
556 uint8_t tx_buff[sizeof(gpi_buffer_)] = {};
557 tx_buff[0] = gpi_buffer_[0];
558 if(transport_.TransmitAndReceive(
559 tx_buff, gpi_buffer_, sizeof(gpi_buffer_))
560 != Transport::Result::OK)
561 {
562 gpi_buffer_[0] = tx_buff[0];
563 return Result::ERR;
564 }
565 gpi_buffer_[0] = tx_buff[0];
566 }
567
568 return Result::OK;
569 }

579 static uint16_t VoltsTo12BitUint(float volts, DacVoltageRange range)
580 {
581 float vmax = 0;
582 float vmin = 0;
583 float vscaler = 0;
584 switch(range)
585 {
586 case DacVoltageRange::NEGATIVE_10_TO_0:
587 vmin = -10;
588 vmax = 0;
589 vscaler = 4095.0f / (vmax - vmin);
590 break;
591 case DacVoltageRange::NEGATIVE_5_TO_5:
592 vmin = -5;
593 vmax = 5;
594 vscaler = 4095.0f / (vmax - vmin);
595 break;
596 case DacVoltageRange::ZERO_TO_10:
597 vmin = 0;
598 vmax = 10;
599 vscaler = 4095.0f / (vmax - vmin);
600 break;
601 }
602 // Clamp...
603 if(volts > vmax)
604 volts = vmax;
605
606 if(volts < vmin)
607 volts = vmin;
608
609 return static_cast<uint16_t>((volts - vmin) * vscaler);
610 }

621 static float TwelveBitUintToVolts(uint16_t value, AdcVoltageRange range)
622 {
623 float vmax = 0;
624 float vmin = 0;
625 float vscaler = 0;
626 switch(range)
627 {
628 case AdcVoltageRange::NEGATIVE_10_TO_0:
629 vmin = -10;
630 vmax = 0;
631 vscaler = (vmax - vmin) / 4095;
632 break;
633 case AdcVoltageRange::NEGATIVE_5_TO_5:
634 vmin = -5;
635 vmax = 5;
636 vscaler = (vmax - vmin) / 4095;
637 break;
638 case AdcVoltageRange::ZERO_TO_10:
639 vmin = 0;
640 vmax = 10;
641 vscaler = (vmax - vmin) / 4095;
642 break;
643 case AdcVoltageRange::ZERO_TO_2P5:
644 vmin = 0;
645 vmax = 2.5;
646 vscaler = (vmax - vmin) / 4095;
647 break;
648 }
649 // Clamp...
650 if(value > 4095)
651 value = 4095;
652
653 return (value * vscaler) + vmin;
654 }

713 Result SetPinConfig(Pin pin)
714 {
715 uint16_t pin_func_cfg = 0x0000;
716
717 if(pin_configurations_[pin].mode != PinMode::NONE)
718 {
719 // Set the pin to high impedance mode before changing (as per the datasheet).
720 WriteRegister(MAX11300_FUNC_BASE + pin, 0x0000);
721 // According to the datasheet, the amount of time necessary for the pin to
722 // switch to high impedance mode depends on the prior configuration.
723 // The worst case recommended wait time seems to be 1ms.
724 DelayUs(1000);
725 }
726
727 // Apply the pin configuration
728 pin_func_cfg = pin_func_cfg
729 | static_cast<uint16_t>(pin_configurations_[pin].mode);
730
731 if(pin_configurations_[pin].mode == PinMode::ANALOG_OUT)
732 {
733 pin_func_cfg
734 |= static_cast<uint16_t>(pin_configurations_[pin].range.dac);
735 }
736 else if(pin_configurations_[pin].mode == PinMode::ANALOG_IN)
737 {
738 // In ADC mode we'll average 128 samples per Update
739 pin_func_cfg
740 = pin_func_cfg | 0x00e0
741 | static_cast<uint16_t>(pin_configurations_[pin].range.adc);
742 }
743 else if(pin_configurations_[pin].mode == PinMode::GPI)
744 {
745 // The DAC data register for that port needs to be set to the value corresponding to the
746 // intended input threshold voltage. Any input voltage above that programmed threshold is
747 // reported as a logic one. The input voltage must be between 0V and 5V.
748 // It may take up to 1ms for the threshold voltage to be effective
749 WriteRegister((MAX11300_DACDAT_BASE + pin),
750 MAX11300Driver::VoltsTo12BitUint(
751 pin_configurations_[pin].threshold,
752 pin_configurations_[pin].range.dac));
753 }
754 else if(pin_configurations_[pin].mode == PinMode::GPO)
755 {
756 // The port’s DAC data register needs to be set first. It may require up to 1ms for the
757 // port to be ready to produce the desired logic one level.
758 WriteRegister((MAX11300_DACDAT_BASE + pin),
759 MAX11300Driver::VoltsTo12BitUint(
760 pin_configurations_[pin].threshold,
761 pin_configurations_[pin].range.dac));
762 }
763
764 // Write the configuration now...
765 if(WriteRegister(MAX11300_FUNC_BASE + pin, pin_func_cfg) != Result::OK)
766 {
767 return Result::ERR;
768 }
769
770 // Wait for 1ms as per the datasheet
771 DelayUs(1000);
772
773 // Verify our configuration was written
774 if(ReadRegister(MAX11300_FUNC_BASE + pin) != pin_func_cfg)
775 {
776 return Result::ERR;
777 }
778
779 // Update and re-index the pin configuration now...
780 UpdatePinConfig();
781
782 return Result::OK;
783 }
784
788 Result UpdatePinConfig()
789 {
790 // Zero everything out...
791 std::memset(dac_buffer_, 0, sizeof(dac_buffer_));
792 std::memset(adc_buffer_, 0, sizeof(adc_buffer_));
793 std::memset(gpi_buffer_, 0, sizeof(gpi_buffer_));
794 std::memset(gpo_buffer_, 0, sizeof(gpo_buffer_));
795
796 dac_pin_count_ = 0;
797 adc_pin_count_ = 0;
798 gpi_pin_count_ = 0;
799 gpo_pin_count_ = 0;
800
801 for(uint8_t i = 0; i <= Pin::PIN_19; i++)
802 {
803 Pin pin = static_cast<Pin>(i);
804
805 // Always reset the value pointer first...
806 pin_configurations_[i].value = nullptr;
807
808 if(pin_configurations_[i].mode == PinMode::ANALOG_OUT)
809 {
810 dac_pin_count_++;
811 if(dac_pin_count_ == 1)
812 {
813 // If this is the first pin of this type, we need to set
814 // the initial address of the dac_buffer_ to point at this pin.
815 // The ordering of subsequent pins is known by the MAX11300.
816 dac_buffer_[0] = (MAX11300_DACDAT_BASE + pin) << 1;
817 }
818 // set the pin_config.value to a pointer at the appropriate
819 // index of the dac_buffer...
820 pin_configurations_[i].value = reinterpret_cast<uint16_t*>(
821 &dac_buffer_[(2 * dac_pin_count_) - 1]);
822 }
823 else if(pin_configurations_[i].mode == PinMode::ANALOG_IN)
824 {
825 adc_pin_count_++;
826 if(adc_pin_count_ == 1)
827 {
828 // If this is the first pin of this type, we need to set
829 // the initial address of the adc_buffer_ to point at this pin.
830 // The ordering of subsequent pins is known by the MAX11300.
831 adc_buffer_[0] = ((MAX11300_ADCDAT_BASE + pin) << 1) | 1;
832 }
833 // set the pin_config.value to a pointer at the appropriate
834 // index of the adc_buffer...
835 pin_configurations_[i].value = reinterpret_cast<uint16_t*>(
836 &adc_buffer_[(2 * adc_pin_count_) - 1]);
837 }
838 else if(pin_configurations_[i].mode == PinMode::GPI)
839 {
840 gpi_pin_count_++;
841 gpi_buffer_[0] = (MAX11300_GPIDAT << 1) | 1;
842 }
843 else if(pin_configurations_[i].mode == PinMode::GPO)
844 {
845 gpo_pin_count_++;
846 gpo_buffer_[0] = (MAX11300_GPODAT << 1);
847 }
848 }
849
850 return Result::OK;
851 }

882 Result ReadRegister(uint8_t address, uint16_t* values, size_t size)
883 {
884 size_t rx_length = (size * 2) + 1;
885 uint8_t rx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
886 uint8_t tx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
887 tx_buff[0] = (address << 1) | 1;
888
889 if(transport_.TransmitAndReceive(tx_buff, rx_buff, rx_length)
890 != Transport::Result::OK)
891 {
892 return Result::ERR;
893 }
894
895 size_t rx_idx = 1;
896 for(size_t i = 0; i < size; i++)
897 {
898 values[i] = static_cast<uint16_t>((rx_buff[rx_idx] << 8)
899 + rx_buff[rx_idx + 1]);
900 rx_idx = rx_idx + 2;
901 }
902 return Result::OK;
903 }

923 Result WriteRegister(uint8_t address, uint16_t* values, size_t size)
924 {
925 size_t tx_size = (size * 2) + 1;
926 uint8_t tx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
927 tx_buff[0] = (address << 1);
928
929 size_t tx_idx = 1;
930 for(size_t i = 0; i < size; i++)
931 {
932 tx_buff[tx_idx++] = static_cast<uint8_t>(values[i] >> 8);
933 tx_buff[tx_idx++] = static_cast<uint8_t>(values[i]);
934 }
935
936 if(transport_.Transmit(tx_buff, tx_size, 0) == Transport::Result::OK)
937 {
938 return Result::OK;
939 }
940
941 return Result::ERR;
942 }
