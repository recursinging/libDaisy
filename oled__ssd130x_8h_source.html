239 void DrawPixel(uint_fast8_t x, uint_fast8_t y, bool on)
240 {
241 if(x >= width || y >= height)
242 return;
243 if(on)
244 buffer_[x + (y / 8) * width] |= (1 << (y % 8));
245 else
246 buffer_[x + (y / 8) * width] &= ~(1 << (y % 8));
247 }

249 void Fill(bool on)
250 {
251 for(size_t i = 0; i < sizeof(buffer_); i++)
252 {
253 buffer_[i] = on ? 0xff : 0x00;
254 }

260 void Update()
261 {
262 uint8_t i;
263 uint8_t high_column_addr;
264 switch(height)
265 {
266 case 32: high_column_addr = 0x12; break;
267
268 default: high_column_addr = 0x10; break;
269 }
270 for(i = 0; i < (height / 8); i++)
271 {
272 transport_.SendCommand(0xB0 + i);
273 transport_.SendCommand(0x00);
274 transport_.SendCommand(high_column_addr);
275 transport_.SendData(&buffer_[width * i], width);
276 }
277 };

279 private:
280 Transport transport_;
281 uint8_t buffer_[width * height / 8];
